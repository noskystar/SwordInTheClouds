```ts
export const UI_FONT_FAMILY = '"PingFang SC", "Microsoft YaHei", "Noto Sans SC", sans-serif';
export const UI_TEXT_RESOLUTION = 4;

type TextStyle = Phaser.Types.GameObjects.Text.TextStyle;

function parseFontSize(fontSize: TextStyle['fontSize']): number {
  if (typeof fontSize === 'number') {
    return fontSize;
  }
  if (typeof fontSize === 'string') {
    const pxMatch = fontSize.match(/^(\d+(?:\.\d+)?)px$/);
    if (pxMatch) {
      return parseFloat(pxMatch[1]);
    }
  }
  // Fallback for em/rem/other units
  return 14;
}

function resolveResolution(fontSize: TextStyle['fontSize']): number {
  const size = parseFontSize(fontSize);
  if (size >= 16) return 4;
  if (size >= 12) return 2;
  return 1;
}

export function uiTextStyle(style: TextStyle): TextStyle {
  const baseResolution = resolveResolution(style.fontSize);
  return {
    fontFamily: UI_FONT_FAMILY,
    resolution: baseResolution,
    padding: { y: 1 },
    ...style,
    // Explicit resolution in style takes precedence
    resolution: style.resolution ?? baseResolution,
  };
}
```

```ts

  this.handleMovement();

```

添加私有方法：
```ts
private updateBob(time: number, delta: number): void {
  const body = this.body as Phaser.Physics.Arcade.Body;

  if (this.isMoving) {
    this.bobTime += delta * this.BOB_SPEED;
    const bobOffset = Math.sin(this.bobTime) * this.BOB_AMPLITUDE;
    this.y = body.y + bobOffset;
  } else {
    // 停止时立即复位，0.5px 跳变不可见
    this.bobTime = 0;
    this.y = body.y;
  }
}
```

**关键说明**：`body.y` 是物理体的位置（脚底），sprite 的 `this.y` 在此基础上叠加 bob 偏移。停止时 `this.y = body.y` 确保完全复位。`BOB_AMPLITUDE` 只用 0.5px，避免碰撞体明显漂移。

```ts
  private updateTeleportHints(): void {
    const px = this.player.x;
    const py = this.player.y;
    const FADE_DISTANCE = 64;

    for (const hint of this.teleportHints) {
      const distance = Phaser.Math.Distance.Between(px, py, hint.cx, hint.cy);
      const targetAlpha = distance < FADE_DISTANCE ? 1 : 0;

      // 每帧以固定速率趋近目标 alpha（约 150ms 淡入/淡出）
      const deltaAlpha = 0.15; // 每帧变化量，假设 60fps 约为 9/frame
      if (hint.text.alpha < targetAlpha) {
        hint.text.setAlpha(Math.min(targetAlpha, hint.text.alpha + deltaAlpha));
      } else if (hint.text.alpha > targetAlpha) {
        hint.text.setAlpha(Math.max(targetAlpha, hint.text.alpha - deltaAlpha));
      }
    }
  }
```
